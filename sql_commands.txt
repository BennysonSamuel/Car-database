CREATE TABLE CarModels (
    car_id INT AUTO_INCREMENT PRIMARY KEY,
    brand_id INT,
    model_name VARCHAR(50) NOT NULL,
    car_type VARCHAR(20),
    price FLOAT(10),
    top_speed VARCHAR(20),
    fuel_type VARCHAR(20),
    FOREIGN KEY (brand_id) REFERENCES Brands(brand_id)
);
